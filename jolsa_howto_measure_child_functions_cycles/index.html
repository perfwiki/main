    Jolsa howto measure child functions cycles

    Example program¶

    • ex.c
    #define FOO(__f)        \
static int __f(int i)   \
{                       \
        while (i--);    \
}

FOO(krava_a);
FOO(krava_b);
FOO(krava_c);
FOO(krava_d);

static void krava_g(void)
{
        krava_a(50);
        krava_b(50);
        krava_c(50);
        krava_d(50);
}

int main(int argc, char **argv)
{
        while (1) {
                krava_a(200);
                krava_b(200);
                krava_c(200);
                krava_d(200);
                krava_g();
        }
}

    ex: ex.o

    • krava_[abcd] functions are taking 25% of the load
